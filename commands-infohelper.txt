--- SERVER CONTROL ---
# Start with wrapper (recommended — enables in-game mode switching)
screen -S csgo /srv/csgo-host/csgo-wrapper.sh competitive
screen -S csgo /srv/csgo-host/csgo-wrapper.sh surf
screen -S csgo /srv/csgo-host/csgo-wrapper.sh dm
screen -S csgo /srv/csgo-host/csgo-wrapper.sh kz
screen -S csgo /srv/csgo-host/csgo-wrapper.sh arena
screen -S csgo /srv/csgo-host/csgo-wrapper.sh retake

# Screen management
screen -r csgo            # reattach
screen -ls                # list screens
Ctrl+A then D             # detach (keeps running)

# Stop cleanly
pkill -9 screen; pkill -9 -f srcds

# Systemd (auto-restart, survives reboot)
sudo systemctl start csgo
sudo systemctl stop csgo
sudo systemctl status csgo
journalctl -u csgo -f

--- IN-GAME COMMANDS ---
!ws / !knife / !gloves    # weapon skins
!rtv                      # vote to change map
!nominate                 # nominate a map
!currentmode              # show current mode + maps remaining
!modes                    # mode vote (admin only)
!modes surf               # direct switch (admin only)
!map de_mirage            # change map (admin only)
!maps                     # map selection menu (admin only)

--- KZ COMMANDS ---
!menu                     # checkpoints menu
!options                  # KZ settings
!r                        # restart run

--- SURF COMMANDS ---
!r                        # restart (teleport to start)
!s                        # go back to stage start
!zones                    # zone setup (admin only)

--- SERVER CONSOLE COMMANDS (inside screen) ---
status                    # connected players + server info
meta list                 # MetaMod plugins
sm plugins list           # SourceMod plugins
changelevel de_mirage     # change map
sm plugins reload modeswitch  # reload plugin
maps *                    # list available maps
